public struct SlidingWindow(T)

Stack of columns.

Example

auto sl = SlidingWindow!double(4, 3);
sl.put([0, 3, 6, 9]);
sl.put([1, 4, 7, 9]);
sl.put([2, 5, 8, 0]);
assert(sl[1, 2] == 7);
assert(sl.matrix[2, 1] == 7);
sl.front.equal([0, 3, 6, 9]);
sl.popFront();
sl.front.equal([1, 4, 7, 9]);
sl.backTransversal == [9, 0];
foreach(i; 0..4)
{
	assert(sl.matrix.ptr != sl.data.ptr);
	sl.put([i,i+4,i+8, 0]);
	sl.popFront();
}
assert(sl.matrix.ptr-1 == sl.data.ptr);
sl.put([6, 7, 0, 3]);
assert(sl.length == 3);
assert(sl.matrix.length == 4);
assert(sl.transversal(1) == [6, 7, 7]);
