Quantile function interface

Example

import std.traits, std.mathspecial;

class NormalQuantile : Quantile!real
{
	real opCall(real x)
	in {
		assert(x >= 0);
		assert(x <= 1);
	}
	body {
		return normalDistributionInverse(x);
	}
}

auto qf = new NormalQuantile;
auto x = qf(0.1);
assert(isNormal(x));
